  {\parindent \z@ \raggedright \normalfont

      \if@mainmatter%%%%% Fix for frontmatter, mainmatter, and backmatter 040920
        \DOCH
      \fi
    \fi
    \interlinepenalty\@M
    \if@mainmatter%%%%% Fix for frontmatter, mainmatter, and backmatter 060424

    \else%
      \DOTIS{#1}%
    \fi
  }}

  \vspace*{0\p@}%
  {\parindent \z@ \raggedright
    \normalfont
    \interlinepenalty\@M

    \vskip 40\p@
  }}
\makeatother
